library(zoo)
library(dygraphs)
tmp <- subdata %>% select(date, normTitleCategory) %>% na.omit
tmp <- tmp %>% 
  mutate(month = format(date, "%m"), year = format(date, "%Y")) %>%
  group_by(month, year, normTitleCategory) %>%
  summarise(Count = n()) %>%
  mutate(day = 01)
tmp <- tmp %>% 
  unite(date, year, month, sep = "-") %>% 
  mutate(date = as.yearmon(date))
